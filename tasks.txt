function applyClass(elementId, className) {
    const divElement = document.getElementById(elementId);
    if (divElement) {
        divElement.classList.remove('default-bg', 'blue-bg', 'red-bg');
        divElement.classList.add(className);
    }
}

function calculateCompoundInterest(principal, rate, years) {
    const decimalRate = rate / 100;
    const amount = principal * Math.pow(1 + decimalRate, years);
    return amount.toFixed(2);
}

const compoundP = 10000;
const compoundR = 5;
const compoundT = 3;
const finalAmount = calculateCompoundInterest(compoundP, compoundR, compoundT);
console.log(`Compound Interest Amount after 3 years: $${finalAmount}`);

function countOccurrences(str, char) {
    let count = 0;
    for (let i = 0; i < str.length; i++) {
        if (str[i] === char) count++;
    }
    return count;
}

function findLargestNumber(arr) {
    let max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > max) max = arr[i];
    }
    return max;
}

const nums2 = [22, 5, 40, 60, 100, 80];
console.log('Largest:', findLargestNumber(nums2));

    { title: 'Card 2', content: 'This is the second card in the row.' },

];

const cardContainer = document.getElementById('cardContainer');

function displayCards(data) {
    if (!cardContainer) return;
    data.forEach(item => {
        const card = document.createElement('div');
        card.className = 'card';
        card.innerHTML = `<h2>${item.title}</h2><p>${item.content}</p>`;
        cardContainer.appendChild(card);
    });
}
displayCards(cardData);

function validateForm(event) {
    event.preventDefault();
    document.querySelectorAll('.error').forEach(e => e.textContent = '');

    const name = document.getElementById('name').value;
    const email = document.getElementById('email').value;
    const contact = document.getElementById('contact').value;
    const dob = document.getElementById('dob').value;
    const designation = document.getElementById('designation').value;
    const files = document.getElementById('files').files;

    let valid = true;

    if (!/^[A-Za-z\s]+$/.test(name)) valid = false;
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) valid = false;
    if (!/^[0-9]{10}$/.test(contact)) valid = false;

    if (!dob) valid = false;
    else {
        const age = new Date().getFullYear() - new Date(dob).getFullYear();
        if (age < 18) valid = false;
    }

    if (designation === '') valid = false;
    if (files.length === 0) valid = false;

    if (valid) alert('Form submitted successfully!');
}
